/* COUNTDOWN */
const launchDate=new Date("2026-04-01T00:00:00").getTime()
setInterval(()=>{
const now=new Date().getTime()
const distance=launchDate-now
const d=Math.floor(distance/(1000*60*60*24))
const h=Math.floor((distance%(1000*60*60*24))/(1000*60*60))
const m=Math.floor((distance%(1000*60*60))/(1000*60))
const s=Math.floor((distance%(1000*60))/1000)
document.getElementById("countdown").innerHTML=d+"D "+h+"H "+m+"M "+s+"S"
},1000)

/* MEME BACKGROUND */
const emojis=["😂","🚀","💎","🐸","🔥","🪙"]
function createEmoji(){
let e=document.createElement("div")
e.className="meme"
e.innerText=emojis[Math.floor(Math.random()*emojis.length)]
e.style.left=Math.random()*100+"vw"
e.style.animationDuration=8+Math.random()*10+"s"
document.getElementById("meme-bg").appendChild(e)
setTimeout(()=>e.remove(),18000)
}
setInterval(createEmoji,400)

/* HOLDER COUNT */
async function loadHolders(){
try{
let r=await fetch(
"https://public-api.solscan.io/token/holders?tokenAddress=YOURTOKEN&limit=1"
)
let j=await r.json()
document.getElementById("holders").innerText=j.total
}catch{
document.getElementById("holders").innerText="Loading..."
}
}
setInterval(loadHolders,20000)
loadHolders()
